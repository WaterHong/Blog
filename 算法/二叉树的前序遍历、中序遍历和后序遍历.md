### 二叉树
输入：
```javascript
const tree = {
    val:2,
    left: {
        val: 3,
        right: {
            val: 2
        }
    },
    right: {
        val: 5
    }
}
```

#### 前序遍历
```javascript
var preorderTraversal = function(root) {
    let res=[];
    preorder(root, res);
    return res;
};

var preorder = function(root,res=[]) {
    if(!root) {
        return [];
    }
    res.push(root.val); //先push，再递归
    preorder(root.left,res);
    preorder(root.right,res);
};
```

#### 中序遍历
```javascript
var preorderTraversal = function(root) {
    let res=[];
    preorder(root, res);
    return res;
};

var preorder = function(root,res=[]) {
    if(!root) {
        return [];
    }
    preorder(root.left,res);
    res.push(root.val);
    preorder(root.right,res);
};
```
